Statistical Testing¶

Test 1¶

In [23]:
from scipy import stats

complaint_types = df['Complaint Type'].unique()


group_1 = df[df['Complaint Type'] == complaint_types[0]]['Request_Closing_Time']
group_2 = df[df['Complaint Type'] == complaint_types[1]]['Request_Closing_Time']

t_test, p_value = stats.ttest_ind(group_1, group_2)

print("T-test result: ", t_test)
print("P-value: ", p_value)

if p_value < 0.05:
    print("Null hypothesis Rejected: There is a significant difference in average closing times between complaint types.")
else:
    print("Failed to reject null hypothesis: There is no significant difference in average closing times between complaint types.")
