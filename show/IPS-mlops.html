for i, col in enumerate(financial_cols):
    ax = axes[i]
    
    ax.hist(df[col], bins=30, color='skyblue', edgecolor='black', alpha=0.7)
    ax.set_title(f'Distribution of {col}')
    ax.set_xlabel(f'{col}')
    if i == 0: ax.set_ylabel('Frequency')
    
    # Show skewness

# Lollipop Chart for Missing Values Percentage

# skip non null colums
df_plot = null_stats_sorted[null_stats_sorted['Null Percentage'] > 1]

fig, ax = plt.subplots(figsize=(16, len(df_plot) * 0.4))

# 3. Create the Lollipop Chart
ax.hlines(y=df_plot.index, xmin=0, xmax=df_plot['Null Percentage'], 
          color='lightgray', linestyle='--', alpha=0.7)

# Plot
colors = ['firebrick' if x > 40 else 'steelblue' for x in df_plot['Null Percentage']]
ax.scatter(df_plot['Null Percentage'], df_plot.index, 
           color=colors, s=100, edgecolors='black', zorder=3)

# Plot Labels
for i, val in enumerate(df_plot['Null Percentage']):
    ax.text(val + 0.8, i, f'{val:.1f}%', va='center', fontsize=10, 
            fontweight='bold', color='black')

# Title and axix labels
ax.set_title('Missing Values Percentage by Column', fontsize=16, pad=20)
ax.set_xlabel('Null Percentage (%)', fontsize=12)
ax.set_ylabel('Columns', fontsize=12)
